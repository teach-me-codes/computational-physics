$$ H = -t \sum_{\langle i,j\rangle, \sigma} (c^\dagger_{i,\sigma} c_{j,\sigma} + \text{H.c.}) + J \sum_{\langle i,j\rangle} (\textbf{S}_i \cdot \textbf{S}_j - \frac{1}{4} n_i n_j) $$
